using DshEtlSearch.Core.Common.Enums;
using DshEtlSearch.Core.Domain;
using FluentAssertions; // Optional: Makes assertions readable, or use Xunit Assert
using Xunit;

namespace DshEtlSearch.UnitTests.Core.Domain
{
    public class DatasetTests
    {
        [Fact]
        public void Constructor_ShouldInitializeIdAndDate()
        {
            // Arrange & Act
            var dataset = new Dataset("doi:10.1234/test");

            // Assert
            Assert.NotEqual(Guid.Empty, dataset.Id);
            Assert.Equal("doi:10.1234/test", dataset.SourceIdentifier);
            Assert.True(dataset.IngestedAt <= DateTime.UtcNow);
        }

        [Fact]
        public void Constructor_ShouldThrowException_WhenIdentifierIsNull()
        {
            // Assert
            Assert.Throws<ArgumentNullException>(() => new Dataset(null));
        }

        [Fact]
        public void AddDocument_ShouldAddDocumentToList_WithCorrectDetails()
        {
            // Arrange
            var dataset = new Dataset("doi:test");
            var fileName = "report.pdf";
            var size = 1024L;

            // Act
            dataset.AddDocument(fileName, FileType.Pdf, size);

            // Assert
            Assert.Single(dataset.Documents); // List should have 1 item
            
            var doc = dataset.Documents[0];
            Assert.Equal(fileName, doc.FileName);
            Assert.Equal(FileType.Pdf, doc.Type);
            Assert.Equal(size, doc.SizeBytes);
            Assert.Equal(dataset.Id, doc.DatasetId); // Verify Foreign Key link
        }
    }
}

using DshEtlSearch.Core.Domain;
using Xunit;

namespace DshEtlSearch.UnitTests.Core.Domain
{
    public class MetadataRecordTests
    {
        [Fact]
        public void ToEmbeddingText_ShouldCombineFieldsCorrectly()
        {
            // Arrange
            var record = new MetadataRecord
            {
                Title = "Flood Risk 2024",
                Abstract = "A detailed study of rainfall.",
                Keywords = "flood, rain, uk"
            };

            // Act
            var result = record.ToEmbeddingText();

            // Assert
            // We expect a specific format for vector generation to ensure consistency
            var expected = "Title: Flood Risk 2024\nAbstract: A detailed study of rainfall.\nKeywords: flood, rain, uk";
            Assert.Equal(expected, result);
        }

        [Fact]
        public void ToEmbeddingText_ShouldHandleNullFields_Gracefully()
        {
            // Arrange
            var record = new MetadataRecord
            {
                Title = "Incomplete Data",
                Abstract = null,
                Keywords = null
            };

            // Act
            var result = record.ToEmbeddingText();

            // Assert
            // Should not throw NullReferenceException
            Assert.Contains("Title: Incomplete Data", result);
        }
    }
}

using DshEtlSearch.Core.Common;
using Xunit;

namespace DshEtlSearch.UnitTests.Core.Domain
{
    public class ResultTests
    {
        [Fact]
        public void Success_ShouldSetIsSuccessTrue_AndContainValue()
        {
            // Act
            var result = Result<int>.Success(42);

            // Assert
            Assert.True(result.IsSuccess);
            Assert.Equal(42, result.Value);
            Assert.Null(result.Error);
        }

        [Fact]
        public void Failure_ShouldSetIsSuccessFalse_AndContainError()
        {
            // Act
            var result = Result<int>.Failure("Something went wrong");

            // Assert
            Assert.False(result.IsSuccess);
            Assert.Equal(default(int), result.Value); // int default is 0
            Assert.Equal("Something went wrong", result.Error);
        }

        [Fact]
        public void VoidSuccess_ShouldSetIsSuccessTrue()
        {
            // Act
            var result = Result.Success();

            // Assert
            Assert.True(result.IsSuccess);
            Assert.Null(result.Error);
        }
    }
